Уменьшение уровня логирования Clickhouse

В файле /etc/clickhouse-server/config.d/set-log-level.xml:

<yandex>
    <logger>
        <level>warning</level>
        <console>true</console>
    </logger>
    <query_thread_log remove="remove"/>
    <query_log remove="remove"/>
    <text_log remove="remove"/>
    <trace_log remove="remove"/>
    <metric_log remove="remove"/>
    <asynchronous_metric_log remove="remove"/>
</yandex>

В новых версиях Clickhouse вместо <yandex> использовать <clickhouse>. Также можно отключить дополнительные логи:

<session_log remove="remove"/>
<part_log remove="remove"/>
